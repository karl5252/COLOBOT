void BotMaker()
{
	object item;
	item=radar(Titanium);
	goto(item.position);
	grab();
	item=radar(BotFactory);
	goto(item.position);
	drop();
	move(-2.5);
	item.factory(TrackedGrabber);
	
	do
	{
		wait(1);              // poczekaj na kostkÄ™ tytanu
		item = radar(TrackedGrabber, 0, 45, 0, 5);
	}
	while ( item == null );
	item=radar(PowerCell);
	goto(item.position);
	grab();
	item=radar(TrackedGrabber);
	goto(item.position);
	drop();
	move(-3);
	message("New Grabber ready!", DisplayInfo);       
}
    


void GrabTit()
{
	object item;
	item = radar(Titanium);
	if (item == null) {
		message("No TITANIUM on planet, waiting", DisplayInfo);
		wait(1);
		
	} else {
		goto(item.position);
		grab();
	}
	
	
}
